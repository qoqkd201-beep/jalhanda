<body>
  <h1>잘한다 잘한다 🎉</h1>
  <button id="poop-btn" onclick="celebrate()">💩</button>
  <div id="message"></div>
  <div id="count">아직 아무것도 안 눌렀어요</div>

  <script>
    let clickCount = 0;
    const messages = [
      "와 진짜 잘한다!! 👏👏👏",
      "최고야!! 역시 믿었어!! 🌟",
      "대단해!! 천재 아니야?? ✨",
      "또 해줘!! 너무 잘해!! 🔥",
      "와 이번엔 더 잘했어!! 🚀",
      "진짜진짜 잘한다!! 💪",
      "너 때문에 행복해!! 🥹",
    ];
    const colors = ['#FF6B6B','#FFD93D','#6BCB77','#4D96FF','#FF922B','#CC5DE8','#20C997'];

    function celebrate() {
      clickCount++;
      const btn = document.getElementById('poop-btn');
      const msg = document.getElementById('message');
      const cnt = document.getElementById('count');
      btn.classList.remove('pop');
      void btn.offsetWidth;
      btn.classList.add('pop');
      msg.textContent = messages[Math.floor(Math.random() * messages.length)];
      msg.classList.remove('show');
      void msg.offsetWidth;
      msg.classList.add('show');
      cnt.textContent = `${clickCount}번 칭찬받음 🏆`;
      launchConfetti();
    }

    function launchConfetti() {
      for (let i = 0; i < 40; i++) {
        const c = document.createElement('div');
        c.className = 'confetti-piece';
        c.style.left = (15 + Math.random() * 70) + '%';
        c.style.top = (5 + Math.random() * 35) + '%';
        c.style.background = colors[Math.floor(Math.random() * colors.length)];
        const size = (8 + Math.random() * 9) + 'px';
        c.style.width = size;
        c.style.height = size;
        c.style.borderRadius = Math.random() > 0.5 ? '50%' : '2px';
        c.style.animationDuration = (0.9 + Math.random() * 1.1) + 's';
        c.style.animationDelay = (Math.random() * 0.25) + 's';
        document.body.appendChild(c);
        c.addEventListener('animationend', () => c.remove());
      }
    }
  </script>
